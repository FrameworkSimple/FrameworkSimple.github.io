108     /**
109      * set up this controller with hooks and names
110      */
111     public function init() {
112 
113         // set the controller to the name of this class
114         self::$controller = get_class($this);
115 
116         // set the controller name to the name of the class minus "Controller"
117         self::$controller_name = strtolower(str_replace("Controller", "", self::$controller));
118 
119         Hook::register("beforeAction",array(get_called_class(),"beforeAction"));
120         Hook::register("afterAction",array(get_called_class(),"afterAction"));
121         Hook::register("beforeRender",array(get_called_class(),"beforeRender"));
122 
123     }
