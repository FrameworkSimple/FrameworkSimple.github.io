 16 /**
 17 * The class provides ability to compare 2 database structure dumps and compile a set of sql statements to update
 18 * one database to make it structure identical to another.
 19 *
 20 * @author Kirill Gerasimenko <ger.kirill@gmail.com>
 21 * @category Extensions
 22 * @package  Extensions
 23 * @subpackage AdminPanel
 24 *
 25 * The input for the script could be taken from the phpMyAdmin structure dump, or provided by some custom code
 26 * that uses 'SHOW CREATE TABLE' query to get database structure table by table.
 27 * The output is either array of sql statements suitable for executions right from php or a string where the
 28 * statements are placed each at new line and delimited with ';' - suitable for execution from phpMyAdmin SQL
 29 * page.
 30 * The resulting sql may contain queries that aim to:
 31 * Create missing table (CREATE TABLE query)
 32 * Delete table which should not longer exist (DROP TABLE query)
 33 * Update, drop or add table field or index definition (ALTER TABLE query)
 34 *
 35 * Some features:
 36 * - AUTO_INCREMENT value is ommited during the comparison and in resulting CREATE TABLE sql
 37 * - fields with definitions like "(var)char (255) NOT NULL default ''" and "(var)char (255) NOT NULL" are treated
 38 *   as equal, the same for (big|tiny)int NOT NULL default 0;
 39 * - IF NOT EXISTS is automatically added to the resulting sql CREATE TABLE statement
 40 * - fields updating queries always come before key modification ones for each table
 41 * Not implemented:
 42 * - The class even does not try to insert or re-order fields in the same order as in the original table.
 43 *   Does order matter?
 44 * IMPORTANT!!! Class will not handle a case when the field was renamed. It will generate 2 queries - one to drop
 45 * the column with the old name and one to create column with the new name, so if there is a data in the dropped
 46 * column, it will be lost.
 47 * Usage example:
 48 *  $updater = new dbStructUpdater();
 49 *  $res = $updater->getupdates($struct1, $struct2);
 50 * -----
 51 * $res == array (
 52 *   [0]=>"ALTER TABLE `b` MODIFY `name` varchar(255) NOT NULL",
 53 *   ...
 54 * )
 55 */

100     * Returns array of update SQL with default options, $source, $dest - database structures
101     * @access public
102     * @param string $source structure dump of database to update
103     * @param string $dest structure dump of the reference database
104     * @param bool $asString if true - result will be a string, otherwise - array

107     function getupdates($source, $dest, $asString=false)
108     {
109         $result = $asString?'':array();
110         $compRes = $this->compare($source, $dest);
111         if (empty($compRes))
112         {
113             return $result;
114         }
115         $compRes = $this->filterdiffs($compRes);
116         if (empty($compRes))
117         {

119         }
120         $result = $this->getdiffsql($compRes);
121         if ($asString)
122         {
123             $result = implode(";\r\n", $result).';';
124         }
125         return $result;

128     /**
129     * Filters comparison result and lefts only sync actions allowed by 'updateTypes' option
130     */
131     function filterdiffs($compRes)
132     {
133         $result = array();
134         if (is_array($this->config['updateTypes']))
135         {
136             $updateActions = $this->config['updateTypes'];
137         }
138         else
139         {
140             $updateActions = array_map('trim', explode(',', $this->config['updateTypes']));
141         }
142         $allowedActions = array('create', 'drop', 'add', 'remove', 'modify');
143         $updateActions = array_intersect($updateActions, $allowedActions);
144         foreach($compRes as $table=>$info)
145         {
146             if ($info['sourceOrphan'])
147             {
148                 if (in_array('create', $updateActions))
149                 {
150                     $result[$table] = $info;
151                 }
152             }
153             elseif ($info['destOrphan'])
154             {
155                 if (in_array('drop', $updateActions))
156                 {
157                     $result[$table] = $info;
158                 }
159             }
160             elseif($info['differs'])
161             {
162                 $resultInfo = $info;
163                 unset($resultInfo['differs']);
164                 foreach ($info['differs'] as $diff)
165                 {
166                     if (empty($diff['dest']) && in_array('add', $updateActions))
167                     {
168                         $resultInfo['differs'][] = $diff;
169                     }
170                     elseif (empty($diff['source']) && in_array('remove', $updateActions))
171                     {
172                         $resultInfo['differs'][] = $diff;
173                     }
174                     elseif(in_array('modify', $updateActions))
175                     {
176                         $resultInfo['differs'][] = $diff;
177                     }
178                 }
179                 if (!empty($resultInfo['differs']))
180                 {
181                     $result[$table] = $resultInfo;
182                 }
183             }
184         }
185         return $result;
186     }

217     /**
218     * Makes comparison of the given database structures, support some options
219     * @access private
220     * @param string $source and $dest are strings - database tables structures
221     * @return array
222     * - table (array)
223     *       - destOrphan (boolean)
224     *       - sourceOrphan (boolean)
225     *       - differs (array) OR (boolean) false if no diffs
226     *           - [0](array)
227     *               - source (string) structure definition line in the out-of-date table
228     *               - dest (string) structure definition line in the reference table
229     *           - [1](array) ...
230     */
231     function compare($source, $dest)
232     {
233         $this->sourceStruct = $source;
234         $this->destStruct = $dest;
235 
236         $result = array();
237         $destTabNames = $this->gettablelist($this->destStruct);
238         $sourceTabNames = $this->gettablelist($this->sourceStruct);
239 
240         $common = array_intersect($destTabNames, $sourceTabNames);
241         $destOrphans = array_diff($destTabNames, $common);
242         $sourceOrphans = array_diff($sourceTabNames, $common);
243         $all = array_unique(array_merge($destTabNames, $sourceTabNames));
244         sort($all);
245         foreach ($all as $tab)
246         {
247             $info = array('destOrphan'=>false, 'sourceOrphan'=>false, 'differs'=>false);
248             if(in_array($tab, $destOrphans))
249             {
250                 $info['destOrphan'] = true;
251             }
252             elseif (in_array($tab, $sourceOrphans))
253             {
254                 $info['sourceOrphan'] = true;
255             }
256             else
257             {
258                 $destSql = $this->gettabsql($this->destStruct, $tab, true);
259                 $sourceSql = $this->gettabsql($this->sourceStruct, $tab, true);
260                 $diffs = $this->comparesql($sourceSql, $destSql);
261                 if ($diffs===false)
262                 {
263                     trigger_error('[WARNING] error parsing definition of table "'.$tab.'" - skipped');
264                     continue;
265                 }
266                 elseif (!empty($diffs))//not empty array
267                 {
268                     $info['differs'] = $diffs;
269                 }
270                 else continue;//empty array
271             }
272             $result[$tab] = $info;
273         }
274         return $result;
275     }

295     /**
296     * Retrieves table structure definition from the database structure dump
297     * @access private
298     * @param string $struct database structure listing
299     * @param string $tab table name
300     * @param bool $removeDatabase - either to remove database name in "CREATE TABLE database.tab"-like declarations
301     * @return string table structure definition
302     */
303     function gettabsql($struct, $tab, $removeDatabase=true)
304     {
305         $result = '';
306         /* create table should be single line in this case*/
307         //1 - part before database, 2-database name, 3 - part after database
308         if (preg_match('/(CREATE(?:\s*TEMPORARY)?\s*TABLE\s*(?:IF NOT EXISTS\s*)?)(?:`?(\w+)`?\.)?(`?('.$tab.')`?(\W|$))/i', $struct, $m, PREG_OFFSET_CAPTURE))
309         {
310             $tableDef = $m[0][0];
311             $start = $m[0][1];
312             $database = $m[2][0];
313             $offset = $start+strlen($m[0][0]);
314             $end = $this->getdelimpos($struct, $offset);
315             if ($end === false)
316             {
317                 $result = substr($struct, $start);
318             }
319             else
320             {
321                 $result = substr($struct, $start, $end-$start);//already without ';'
322             }
323         }
324         $result = trim($result);
325         if ($database && $removeDatabase)
326         {
327             $result = str_replace($tableDef, $m[1][0].$m[3][0], $result);
328         }
329         return $result;
330     }

332     /**
333     * Splits table sql into indexed array
334     *
335     */
336     function splittabsql($sql)
337     {
338         $result = array();
339         //find opening bracket, get the prefix along with it
340         $openBracketPos = $this->getdelimpos($sql, 0, '(');
341         if ($openBracketPos===false)
342         {
343             trigger_error('[WARNING] can not find opening bracket in table definition');
344             return false;
345         }
346         $prefix = substr($sql, 0, $openBracketPos+1);//prefix can not be empty, so do not check it, just trim
347         $result[] = trim($prefix);
348         $body = substr($sql, strlen($prefix));//fields, indexes and part after closing bracket
349         //split by commas, get part by part
350         while(($commaPos = $this->getdelimpos($body, 0, ',', true))!==false)
351         {
352             $part = trim(substr($body, 0, $commaPos+1));//read another part and shorten $body
353             if ($part)
354             {
355                 $result[] = $part;
356             }
357             $body = substr($body, $commaPos+1);
358         }
359         //here we have last field (or index) definition + part after closing bracket (ENGINE, ect)
360         $closeBracketPos = $this->getdelimrpos($body, 0, ')');
361         if ($closeBracketPos===false)
362         {
363             trigger_error('[WARNING] can not find closing bracket in table definition');
364             return false;
365         }
366         //get last field / index definition before closing bracket
367         $part = substr($body, 0, $closeBracketPos);
368         $result[] = trim($part);
369         //get the suffix part along with the closing bracket
370         $suffix = substr($body, $closeBracketPos);
371         $suffix = trim($suffix);
372         if ($suffix)
373         {
374             $result[] = $suffix;
375         }
376         return $result;
377     }

379     /**
380     * returns array of fields or keys definitions that differs in the given tables structure
381     * @access private
382     * @param sring $sourceSql table structure
383     * @param sring $destSql right table structure
384     * supports some $options
385     * @return array
386     *   - [0]
387     *       - source (string) out-of-date table field definition
388     *       - dest (string) reference table field definition
389     *   - [1]...
390     */
391     function comparesql($sourceSql, $destSql)//$sourceSql, $destSql
392     {
393         $result = array();
394         //split with comma delimiter, not line breaks
395         $sourceParts =  $this->splittabsql($sourceSql);
396         if ($sourceParts===false)//error parsing sql
397         {
398             trigger_error('[WARNING] error parsing source sql');
399             return false;
400         }
401         $destParts = $this->splittabsql($destSql);
402         if ($destParts===false)
403         {
404             trigger_error('[WARNING] error parsing destination sql');
405             return false;
406         }
407         $sourcePartsIndexed = array();
408         $destPartsIndexed = array();
409         foreach($sourceParts as $line)
410         {
411             $lineInfo = $this->processline($line);
412             if (!$lineInfo) continue;
413             $sourcePartsIndexed[$lineInfo['key']] = $lineInfo['line'];
414         }
415         foreach($destParts as $line)
416         {
417             $lineInfo = $this->processline($line);
418             if (!$lineInfo) continue;
419             $destPartsIndexed[$lineInfo['key']] = $lineInfo['line'];
420         }
421         $sourceKeys = array_keys($sourcePartsIndexed);
422         $destKeys = array_keys($destPartsIndexed);
423         $all = array_unique(array_merge($sourceKeys, $destKeys));
424         sort($all);//fields first, then indexes - because fields are prefixed with '!'
425 
426         foreach ($all as $key)
427         {
428             $info = array('source'=>'', 'dest'=>'');
429             $inSource= in_array($key, $sourceKeys);
430             $inDest= in_array($key, $destKeys);
431             $sourceOrphan = $inSource && !$inDest;
432             $destOrphan = $inDest && !$inSource;
433             $different =  $inSource && $inDest &&
434             strcasecmp($this->normalizestring($destPartsIndexed[$key]), $this->normalizestring($sourcePartsIndexed[$key]));
435             if ($sourceOrphan)
436             {
437                 $info['source'] = $sourcePartsIndexed[$key];
438             }
439             elseif ($destOrphan)
440             {
441                 $info['dest'] = $destPartsIndexed[$key];
442             }
443             elseif ($different)
444             {
445                 $info['source'] = $sourcePartsIndexed[$key];
446                 $info['dest'] = $destPartsIndexed[$key];
447             }
448             else continue;
449             $result[] = $info;
450         }
451         return $result;
452     }

454     /**
455     * Transforms table structure defnition line into key=>value pair where the key is a string that uniquely
456     * defines field or key desribed
457     * @access private
458     * @param string $line field definition string
459     * @return array array with single key=>value pair as described in the description
460     * implements some options
461     */
462     function processline($line)
463     {
464         $options = $this->config;
465         $result = array('key'=>'', 'line'=>'');
466         $line = rtrim(trim($line), ',');
467         if (preg_match('/^(CREATE\s+TABLE)|(\) ENGINE=)/i', $line))//first or last table definition line
468         {
469             return false;
470         }
471         //if (preg_match('/^(PRIMARY KEY)|(((UNIQUE )|(FULLTEXT ))?KEY `?\w+`?)/i', $line, $m))//key definition
472         if (preg_match('/^(PRIMARY\s+KEY)|(((UNIQUE\s+)|(FULLTEXT\s+))?KEY\s+`?\w+`?)/i', $line, $m))//key definition
473         {
474             $key = $m[0];
475         }
476         elseif (preg_match('/^`?\w+`?/i', $line, $m))//field definition
477         {
478             $key = '!'.$m[0];//to make sure fields will be synchronised before the keys
479         }
480         else
481         {
482             return false;//line has no valuable info (empty or comment)
483         }
484         //$key = str_replace('`', '', $key);
485         if (!empty($options['varcharDefaultIgnore']))
486         {
487             $line = preg_replace("/(var)?char\(([0-9]+)\)\s+NOT\s+NULL\s+default\s+''/i", '$1char($2) NOT NULL', $line);
488         }
489         if (!empty($options['intDefaultIgnore']))
490         {
491             $line = preg_replace("/((?:big)|(?:tiny))?int\(([0-9]+)\)\s+NOT\s+NULL\s+default\s+'0'/i", '$1int($2) NOT NULL', $line);
492         }
493         if (!empty($options['ignoreIncrement']))
494         {
495             $line = preg_replace("/ AUTO_INCREMENT=[0-9]+/i", '', $line);
496         }
497         $result['key'] = $this->normalizestring($key);
498         $result['line']= $line;
499         return $result;
500     }

502     /**
503     * Takes an output of compare() method to generate the set of sql needed to update source table to make it
504     * look as a destination one
505     * @access private
506     * @param array $diff compare() method output
507     * @return array list of sql statements
508     * supports query generation options
509     */
510     function getdiffsql($diff)//maybe add option to ommit or force 'IF NOT EXISTS', skip autoincrement
511     {
512         $options = $this->config;
513         $sqls = array();
514         if (!is_array($diff) || empty($diff))
515         {
516             return $sqls;
517         }
518         foreach($diff as $tab=>$info)
519         {
520             if ($info['sourceOrphan'])//delete it
521             {
522                 $sqls[] = 'DROP TABLE `'.$tab.'`';
523             }
524             elseif ($info['destOrphan'])//create destination table in source
525             {
526                 $database = '';
527                 $destSql = $this->gettabsql($this->destStruct, $tab, $database);
528                 if (!empty($options['ignoreIncrement']))
529                 {
530                     $destSql = preg_replace("/\s*AUTO_INCREMENT=[0-9]+/i", '', $destSql);
531                 }
532                 if (!empty($options['ingoreIfNotExists']))
533                 {
534                     $destSql = preg_replace("/IF NOT EXISTS\s*/i", '', $destSql);
535                 }
536                 if (!empty($options['forceIfNotExists']))
537                 {
538                     $destSql = preg_replace('/(CREATE(?:\s*TEMPORARY)?\s*TABLE\s*)(?:IF\sNOT\sEXISTS\s*)?(`?\w+`?)/i', '$1IF NOT EXISTS $2', $destSql);
539                 }
540                 $sqls[] = $destSql;
541             }
542             else
543             {
544                 foreach($info['differs'] as $finfo)
545                 {
546                     $inDest = !empty($finfo['dest']);
547                     $inSource = !empty($finfo['source']);
548                     if ($inSource && !$inDest)
549                     {
550                         $sql = $finfo['source'];
551                         $action = 'drop';
552                     }
553                     elseif ($inDest && !$inSource)
554                     {
555                         $sql = $finfo['dest'];
556                         $action = 'add';
557                     }
558                     else
559                     {
560                         $sql = $finfo['dest'];
561                         $action = 'modify';
562                     }
563                     $sql = $this->getactionsql($action, $tab, $sql);
564                     $sqls[] = $sql;
565                 }
566             }
567         }
568         return $sqls;
569     }

571     /**
572     * Compiles update sql
573     * @access private
574     * @param string $action - 'drop', 'add' or 'modify'
575     * @param string $tab table name
576     * @param string $sql definition of the element to change
577     * @return string update sql
578     */
579     function getactionsql($action, $tab, $sql)
580     {
581         $result = 'ALTER TABLE `'.$tab.'` ';
582         $action = strtolower($action);
583         $keyField = '`?\w`?(?:\(\d+\))?';//matches `name`(10)
584         $keyFieldList = '(?:'.$keyField.'(?:,\s?)?)+';//matches `name`(10),`desc`(255)
585         if (preg_match('/((?:PRIMARY )|(?:UNIQUE )|(?:FULLTEXT ))?KEY `?(\w+)?`?\s(\('.$keyFieldList.'\))/i', $sql, $m))
586         {   //key and index operations
587             $type = strtolower(trim($m[1]));
588             $name = trim($m[2]);
589             $fields = trim($m[3]);
590             switch($action)
591             {
592                 case 'drop':
593                     if ($type=='primary')
594                     {
595                         $result.= 'DROP PRIMARY KEY';
596                     }
597                     else
598                     {
599                         $result.= 'DROP INDEX `'.$name.'`';
600                     }
601                 break;
602                 case 'add':
603                     if ($type=='primary')
604                     {
605                         $result.= 'ADD PRIMARY KEY '.$fields;
606                     }
607                     elseif ($type=='')
608                     {
609                         $result.= 'ADD INDEX `'.$name.'` '.$fields;
610                     }
611                     else
612                     {
613                         $result .='ADD '.strtoupper($type).' `'.$name.'` '.$fields;//fulltext or unique
614                     }
615                 break;
616                 case 'modify':
617                     if ($type=='primary')
618                     {
619                         $result.='DROP PRIMARY KEY, ADD PRIMARY KEY '.$fields;
620                     }
621                     elseif ($type=='')
622                     {
623                         $result.='DROP INDEX `'.$name.'`, ADD INDEX `'.$name.'` '.$fields;
624                     }
625                     else
626                     {
627                         $result.='DROP INDEX `'.$name.'`, ADD '.strtoupper($type).' `'.$name.'` '.$fields;//fulltext or unique
628                     }
629                 break;
630 
631             }
632         }
633         else //fields operations
634         {
635             $sql = rtrim(trim($sql), ',');
636             $result.= strtoupper($action);
637             if ($action=='drop')
638             {
639                 $spacePos = strpos($sql, ' ');
640                 $result.= ' '.substr($sql, 0, $spacePos);
641             }
642             else
643             {
644                 $result.= ' '.$sql;
645             }
646         }
647         return $result;
648     }

650     /**
651     * Searches for the position of the next delimiter which is not inside string literal like 'this ; ' or
652     * like "this ; ".
653     *
654     * Handles escaped \" and \'. Also handles sql comments.
655     * Actualy it is regex-based Finit State Machine (FSN)
656     */
657     function getdelimpos($string, $offset=0, $delim=';', $skipInBrackets=false)
658     {
659         $stack = array();
660         $rbs = '\\\\';  //reg - escaped backslash
661         $regPrefix = "(?<!$rbs)(?:$rbs{2})*";
662         $reg = $regPrefix.'("|\')|(/\\*)|(\\*/)|(-- )|(\r\n|\r|\n)|';
663         if ($skipInBrackets)
664         {
665             $reg.='(\(|\))|';
666         }
667         else
668         {
669             $reg.='()';
670         }
671         $reg .= '('.preg_quote($delim).')';
672         while (preg_match('%'.$reg.'%', $string, $m, PREG_OFFSET_CAPTURE, $offset))
673         {
674             $offset = $m[0][1]+strlen($m[0][0]);
675             if (end($stack)=='/*')
676             {
677                 if (!empty($m[3][0]))
678                 {
679                     array_pop($stack);
680                 }
681                 continue;//here we could also simplify regexp
682             }
683             if (end($stack)=='-- ')
684             {
685                 if (!empty($m[5][0]))
686                 {
687                     array_pop($stack);
688                 }
689                 continue;//here we could also simplify regexp
690             }
691 
692             if (!empty($m[7][0]))// ';' found
693             {
694                 if (empty($stack))
695                 {
696                     return $m[7][1];
697                 }
698                 else
699                 {
700                     //var_dump($stack, substr($string, $offset-strlen($m[0][0])));
701                 }
702             }
703             if (!empty($m[6][0]))// '(' or ')' found
704             {
705                 if (empty($stack) && $m[6][0]=='(')
706                 {
707                     array_push($stack, $m[6][0]);
708                 }
709                 elseif($m[6][0]==')' && end($stack)=='(')
710                 {
711                     array_pop($stack);
712                 }
713             }
714             elseif (!empty($m[1][0]))// ' or " found
715             {
716                 if (end($stack)==$m[1][0])
717                 {
718                     array_pop($stack);
719                 }
720                 else
721                 {
722                     array_push($stack, $m[1][0]);
723                 }
724             }
725             elseif (!empty($m[2][0])) // opening comment / *
726             {
727                 array_push($stack, $m[2][0]);
728             }
729             elseif (!empty($m[4][0])) // opening comment --
730             {
731                 array_push($stack, $m[4][0]);
732             }
733         }
734         return false;
735     }

737     /**
738     * works the same as getDelimPos except returns position of the first occurence of the delimiter starting from
739     * the end of the string
740     */
741     function getdelimrpos($string, $offset=0, $delim=';', $skipInBrackets=false)
742     {
743         $pos = $this->getdelimpos($string, $offset, $delim, $skipInBrackets);
744         if ($pos===false)
745         {
746             return false;
747         }
748         do
749         {
750             $newPos=$this->getdelimpos($string, $pos+1, $delim, $skipInBrackets);
751             if ($newPos !== false)
752             {
753                 $pos = $newPos;
754             }
755         }
756         while($newPos!==false);
757         return $pos;
758     }
